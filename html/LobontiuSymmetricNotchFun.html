function [Disp,K] = LobontiuSymmetricNotchFun(E,t,l,w,F)
% E = Young's Modulus
% t = thickness of center beam
% w = Depth of flexure
% l = Length of beam
% F = Force applied

t=t/5;
%Thickness of thin-member reduced by factor of 10

C = (4*l^3)/(E*w*t^3);
%Compliance

Disp = F*C;
%Displacement
K=1/C;
%Stiffness
end
